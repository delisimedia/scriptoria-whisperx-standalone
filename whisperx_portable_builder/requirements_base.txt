# Base requirements for WhisperX 3.8.2 compatibility with Python 3.13.3
# PyTorch 2.8.0 is officially supported by WhisperX 3.8.x
# These pins prevent dependency drift to incompatible newer versions

# NOTE: whisperx itself is installed separately with --no-deps before this file
# is processed, because whisperx's torch~=2.8.0 pin conflicts with torch 2.9.x
# which is required since cu128 wheels only exist from 2.9.0+.

# SpeechBrain - MUST use 0.5.x to avoid infinite recursion bug in 1.0.x
# (SpeechBrain 1.0.x has AttributeError recursion on Python 3.13 + Windows)
speechbrain<1.0.0

# NOTE: pytorch-lightning and torchmetrics pins removed for pyannote v4 compatibility.
# pyannote.audio>=4.0.0 requires torchmetrics>=1.6.1 and lightning>=2.4 (the new
# unified 'lightning' package), which conflict with the old pytorch-lightning<2.0.0
# and torchmetrics<1.0.0 pins we used for SpeechBrain 0.5.x.
# Let pip resolve these naturally from speechbrain and pyannote's own requirements.

# Pyannote - WhisperX 3.8.x requires pyannote-audio v4 (breaking change from v3)
pyannote.audio>=4.0.0

# Explicitly pin NumPy to avoid issues with Python 3.13
numpy>=2.1.0

# Other dependencies - let WhisperX pull compatible versions
faster-whisper>=1.1.1
transformers>=4.48.0
